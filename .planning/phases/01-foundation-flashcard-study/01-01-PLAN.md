    **src/types/index.ts** — Define all shared types for the flashcard feature:

    ```typescript
    export type CardType = 'definition' | 'enumeration'
    export type ProgressStatus = 'known' | 'learning' | null

    export interface Flashcard {
      id: string
      type: CardType
      term: string
      definition: string       // for definition type
      items?: string[]         // for enumeration type (the list items)
      itemCount?: number       // derived: items.length, used for hint display
    }
    ```

    **src/stores/flashcardStore.ts** — Zustand store with persist + skipHydration:
    ```typescript
    'use client'
    import { create } from 'zustand'
    import { persist } from 'zustand/middleware'
    import { ProgressStatus } from '@/types'

    interface FlashcardState {
      progress: Record<string, ProgressStatus>
      markKnown: (id: string) => void
      markLearning: (id: string) => void
      resetProgress: () => void
      getKnownCount: () => number
      getLearningCount: () => number
    }

    export const useFlashcardStore = create<FlashcardState>()(
      persist(
        (set, get) => ({
          progress: {},
          markKnown: (id) => set((state) => ({ progress: { ...state.progress, [id]: 'known' } })),
          markLearning: (id) => set((state) => ({ progress: { ...state.progress, [id]: 'learning' } })),
          resetProgress: () => set({ progress: {} }),
          getKnownCount: () => Object.values(get().progress).filter(v => v === 'known').length,
          getLearningCount: () => Object.values(get().progress).filter(v => v === 'learning').length,
        }),
        {
          name: 'infoman-flashcard-progress',
          skipHydration: true,   // CRITICAL: prevents SSR/client hydration mismatch
        }
      )
    )
    ```

    **src/components/StorageHydration.tsx** — triggers rehydration after first client render:
    ```typescript
    'use client'
    import { useEffect } from 'react'
    import { useFlashcardStore } from '@/stores/flashcardStore'

    export function StorageHydration() {
      useEffect(() => {
        useFlashcardStore.persist.rehydrate()
      }, [])
      return null
    }
    ```
